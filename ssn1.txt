from django.db import connections

def view_(student_roll_number):
    sql = "select * from student where student_roll_number="+str(student_roll_number)+";"
    cur = connections['default'].cursor()
    cur.execute(sql)
    row_headers = [x[0] for x in cur.description]  # this will extract row headers
    rv = cur.fetchall()
    return dict(genres=[list(i) for i in rv])

def insert():

from django.contrib import messages

def login_(request):
    if request.method == 'POST':
        username = request.POST['username']
        password = request.POST['password']
        user = authenticate(username=username, password=password)
        # user = authenticate(username=username, password=password)
        # print(username,"----p----",password,user)
        
        if user is not None:
            if user.is_active:
                login(request,user)
                if(request.user.groups.filter(name='faculty').exists()):
                    return HttpResponseRedirect('/faculty')
                elif(request.user.groups.filter(name='students').exists()):
                    return HttpResponseRedirect('/student')
        else:
            messages.error(request,'Id/Password Incorrect')
            return redirect('login')
    return render(request, 'registration/login.html')
